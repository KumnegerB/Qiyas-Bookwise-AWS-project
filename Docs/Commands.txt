# Create Books table
aws dynamodb create-table \
    --table-name Books \
    --attribute-definitions AttributeName=bookId,AttributeType=S \
    --key-schema AttributeName=bookId,KeyType=HASH \
    --billing-mode PAY_PER_REQUEST

# Create Reviews table (composite key)
aws dynamodb create-table \
    --table-name Reviews \
    --attribute-definitions \
        AttributeName=bookId,AttributeType=S \
        AttributeName=reviewId,AttributeType=S \
    --key-schema \
        AttributeName=bookId,KeyType=HASH \
        AttributeName=reviewId,KeyType=RANGE \
    --billing-mode PAY_PER_REQUEST

# Insert sample books
aws dynamodb put-item --table-name Books --item '{
    "bookId": {"S": "book-001"},
    "title": {"S": "Serverless on AWS"},
    "author": {"S": "Jane Doe"},
    "description": {"S": "A comprehensive guide to serverless architecture"}
}'

# Query all books
aws dynamodb scan --table-name Books

# Query reviews for a book
aws dynamodb query \
    --table-name Reviews \
    --key-condition-expression "bookId = :bookId" \
    --expression-attribute-values '{":bookId": {"S": "book-001"}}'

    IAM Role Lambda execution

    # Create trust policy
cat > trust-policy.json << 'EOF'
{
  "Version": "2012-10-17",
  "Statement": [{
    "Effect": "Allow",
    "Principal": {"Service": "lambda.amazonaws.com"},
    "Action": "sts:AssumeRole"
  }]
}
EOF

# Create IAM role
aws iam create-role \
    --role-name BookWiseLambdaRole \
    --assume-role-policy-document file://trust-policy.json

# Attach DynamoDB permissions
aws iam put-role-policy \
    --role-name BookWiseLambdaRole \
    --policy-name DynamoDBAccess \
    --policy-document '{
  "Version": "2012-10-17",
  "Statement": [
    {"Effect": "Allow","Action": ["dynamodb:GetItem","dynamodb:Scan","dynamodb:Query","dynamodb:PutItem"],"Resource": ["arn:aws:dynamodb:us-east-1:*:table/Books","arn:aws:dynamodb:us-east-1:*:table/Reviews"]},
    {"Effect": "Allow","Action": ["logs:CreateLogGroup","logs:CreateLogStream","logs:PutLogEvents"],"Resource": "*"}
  ]
}'

    --pool-name BookWiseUsers \
    --query 'UserPool.Id' \

    --output text)

    Api gate way
    # Create API
API_ID=$(aws apigateway create-rest-api --name BookWiseAPI --query 'id' --output text)

# Get root ID
ROOT_ID=$(aws apigateway get-resources --rest-api-id $API_ID --query 'items[0].id' --output text)

# Create /books
BOOKS_ID=$(aws apigateway create-resource --rest-api-id $API_ID --parent-id $ROOT_ID --path-part books --query 'id' --output text)

# Create /books/{bookId}
BOOK_ID=$(aws apigateway create-resource --rest-api-id $API_ID --parent-id $BOOKS_ID --path-part {bookId} --query 'id' --output text)

# Create /reviews
REVIEWS_ID=$(aws apigateway create-resource --rest-api-id $API_ID --parent-id $ROOT_ID --path-part reviews --query 'id' --output text)

# Add GET method to /books
aws apigateway put-method --rest-api-id $API_ID --resource-id $BOOKS_ID --http-method GET --authorization-type NONE
aws apigateway put-integration --rest-api-id $API_ID --resource-id $BOOKS_ID --http-method GET --type AWS_PROXY --integration-http-method POST --uri arn:aws:apigateway:us-east-1:lambda:path/2015-03-31/functions/arn:aws:lambda:us-east-1:926274061952:function:getBooks/invocations
